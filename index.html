Flight Delay Insurance with Real-time API
# Initialize the ModRoll client
from modroll import ModRollClient, Transaction, DataFeed

client = ModRollClient(
    connection_type="sui",
    network="testnet",
    fallback_enabled=True
)

# Flight delay insurance function
def process_flight_insurance(policy_id, flight_number):
    # Connect to flight status API
    flight_api = DataFeed("aviation", {
        "flight": flight_number,
        "api_key": "YOUR_API_KEY"
    })
    
    # Get current flight status data
    flight_status = flight_api.get_data()
    print(f"Processing insurance for flight {flight_number}")
    
    # Calculate compensation based on delay
    delay_minutes = flight_status.get("delay_minutes", 0)
    is_cancelled = flight_status.get("status") == "cancelled"
    
    compensation = 0
    if is_cancelled:
        compensation = 500  # Full compensation for cancellation
    elif delay_minutes >= 180:  # 3+ hours
        compensation = 300
    elif delay_minutes >= 120:  # 2+ hours
        compensation = 200
    elif delay_minutes >= 60:   # 1+ hour
        compensation = 100
    elif delay_minutes >= 30:   # 30+ minutes
        compensation = 50
    
    # Create transaction only if compensation is due
    if compensation > 0:
        transaction = Transaction()
        transaction.add_instruction("process_claim", {
            "policy_id": policy_id,
            "flight_number": flight_number,
            "delay_minutes": delay_minutes,
            "compensation_amount": compensation,
            "is_cancelled": is_cancelled
        })
        
        # Execute on SUI blockchain with automatic failover
        result = client.execute_transaction(transaction)
        print(f"Claim processed with compensation: {compensation}")
        return result
    else:
        print("No compensation due - flight on time or minimal delay")
        return {"status": "no_action_required"}

DeFi Stop-Loss Order with Market Data
// Import ModRoll library
const { ModRollClient, Transaction, DataFeed } = require('modroll');

// Create client with SUI connection and failover
const client = new ModRollClient({
  connectionType: 'sui',
  network: 'mainnet',
  fallbackEnabled: true
});

// Define a stop-loss order function
async function createStopLossOrder(token, amount, stopPrice) {
  // Connect to real-time price feed
  const priceFeed = new DataFeed('crypto_price', {
    token: token,
    source: 'binance'
  });
  
  // Get current token price
  const currentPrice = await priceFeed.getData();
  console.log(`Current price for ${token}: ${currentPrice.price}`);
  
  // Check if stop loss should trigger immediately
  if (currentPrice.price <= stopPrice) {
    console.log(`Stop price ${stopPrice} reached, executing immediately`);
    return executeOrder(token, amount);
  }
  
  // Set up conditional transaction
  const transaction = new Transaction();
  
  // Add price condition
  transaction.addCondition({
    type: 'price_below',
    token: token,
    threshold: stopPrice,
    data_source: 'binance'
  });
  
  // Add sell order instruction
  transaction.addInstruction('market_sell', {
    token: token,
    amount: amount
  });
  
  // Deploy conditional transaction to SUI blockchain
  const result = await client.deployConditionalTransaction(transaction);
  console.log(`Stop-loss order deployed with ID: ${result.order_id}`);
  
  return {
    status: 'pending',
    order_id: result.order_id,
    transaction_id: result.transaction_id
  };
}

Supply Chain Verification on SUI
module product::supply_chain_verifier {
    use sui::object::{Self, UID};
    use sui::transfer;
    use sui::tx_context::{Self, TxContext};
    use std::string::{Self, String};
    use sui::event;
    
    // Product struct representing a physical item
    struct Product has key, store {
        id: UID,
        name: String,
        manufacturer: address,
        manufacture_date: u64,
        batch_id: String,
        current_location: String,
        temperature_log: vector,
        verified: bool
    }
    
    // Event emitted when product location is updated
    struct LocationUpdate has copy, drop {
        product_id: ID,
        new_location: String,
        temperature: u64,
        timestamp: u64
    }
    
    // Event emitted when product is verified
    struct ProductVerified has copy, drop {
        product_id: ID,
        passed: bool,
        verification_time: u64
    }
    
    // Update product location (called from ModRoll with IoT data)
    public fun update_location(
        product: &mut Product,
        new_location: String,
        current_temperature: u64,
        ctx: &mut TxContext
    ) {
        // Update product location
        product.current_location = new_location;
        
        // Record temperature in log
        vector::push_back(&mut product.temperature_log, current_temperature);
        
        // Emit location update event
        event::emit(LocationUpdate {
            product_id: object::id(product),
            new_location,
            temperature: current_temperature,
            timestamp: tx_context::epoch(ctx)
        });
    }
    
    // Verify the product meets quality standards
    public fun verify_product(
        product: &mut Product,
        ctx: &mut TxContext
    ) {
        // Check temperature log for any values outside acceptable range
        let i = 0;
        let temp_ok = true;
        let len = vector::length(&product.temperature_log);
        
        while (i < len) {
            let temp = *vector::borrow(&product.temperature_log, i);
            if (temp < 2 || temp > 8) {  // 2-8°C range for pharmaceuticals
                temp_ok = false;
                break;
            };
            i = i + 1;
        };
        
        // Update product verification status
        product.verified = temp_ok;
        
        // Emit verification event
        event::emit(ProductVerified {
            product_id: object::id(product),
            passed: temp_ok,
            verification_time: tx_context::epoch(ctx)
        });
    }
}
